            dodawana = gen.nextInt(25);
            dodawana = dodawana+65;}
        else if (los > 2 && los <10){
            dodawana = gen.nextInt(25);
            dodawana = dodawana+97;}
        else{
            dodawana = 32;}
        wyraz = wyraz + (char)dodawana;
     }
    
    
    return wyraz;
    }
